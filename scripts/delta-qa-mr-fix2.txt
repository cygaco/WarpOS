function isLinkedInJobUrl(url: string): boolean {
  try {
    const parsed = new URL(url);
    return (
      parsed.hostname.includes("linkedin.com") &&
      parsed.pathname.includes("/jobs/view/")
    );
  } catch {
    return false;
  }
}

export function StepCollect({
  session,
  substep,
  analysisStatus = "idle",
  analysisError,
  jobCount,
  onScrapeComplete,
  onViewTargets,
  onSessionUpdate,
}: StepCollectProps) {
  const [queries, setQueries] = useState<string[]>(
    session.generatedQueries ?? [],
  );
  const [editingIdx, setEditingIdx] = useState<number | null>(null);
  const [editValue, setEditValue] = useState("");
  const [newQuery, setNewQuery] = useState("");
  const [avoidTerms, setAvoidTerms] = useState<
    Array<{ term: string; reason: string } | string>
  >(session.profile?.avoidTerms ?? []);

  const [generatingQueries, setGeneratingQueries] = useState(false);
  const [genError, setGenError] = useState<string | null>(null);

  const [scraping, setScraping] = useState(false);
  const [scrapeError, setScrapeError] = useState<string | null>(null);
  const [scrapePhaseIdx, setScrapePhaseIdx] = useState(0);
  const [scrapePending, setScrapePending] = useState(0);
  const [scrapeTotal, setScrapeTotal] = useState(0);
  const [progressStage, setProgressStage] = useState<string>("queued");
  const [queueCooldownUntil, setQueueCooldownUntil] = useState<number | null>(
    null,
  );

  const [queryStats, setQueryStats] = useState<
    Array<{ query: string; resultCount: number }>
  >(session.queryStats ?? []);

  // Manual URL input (GS-MKT-21 — Post-MVP, wired but lightweight)
  const [manualUrl, setManualUrl] = useState("");
  const [manualUrlError, setManualUrlError] = useState<string | null>(null);

  const scrapePhaseTimerRef = useRef<ReturnType<typeof setInterval> | null>(
    null,
  );
  const pollTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
  const activeTicketRef = useRef<string | null>(null);

  // ── Auto-generate queries on first entry ──────────────────────────────────

  useEffect(() => {
    if (queries.length === 0 && !generatingQueries && session.profile) {
      void generateQueries();
    }
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []);

  // ── Cleanup timers on unmount ─────────────────────────────────────────────

  useEffect(() => {
    return () => {
      if (scrapePhaseTimerRef.current) {
        clearInterval(scrapePhaseTimerRef.current);
      }
      if (pollTimerRef.current) {
        clearTimeout(pollTimerRef.current);
      }
    };
  }, []);

  // ── Mount-recovery: resume polling if a ticket was in-flight ─────────────

  useEffect(() => {
    const savedTicketId = localStorage.getItem("mkt_activeTicketId");
    if (savedTicketId && !scraping) {
      setScraping(true);
      setScrapeError(null);
      activeTicketRef.current = savedTicketId;
      void pollTicket(savedTicketId);
    }
    return () => {
      if (pollTimerRef.current) clearTimeout(pollTimerRef.current);
    };
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []);

  // ── Ticket polling (v3 async model) ──────────────────────────────────────

  const pollTicket = useCallback(
    async (ticketId: string) => {
      const POLL_INTERVAL = Math.floor(Math.random() * 2000) + 3000; // 3–5s jitter per GS-MKT-30

      try {
        const res = await fetch(`${API_BASE}/tickets/${ticketId}`);
        if (!res.ok) {
          if (res.status === 404) {
            setScrapeError("Scrape ticket not found. Please try again.");
            setScraping(false);
            activeTicketRef.current = null;
            localStorage.removeItem("mkt_activeTicketId");
            return;
          }
          // Non-terminal error — keep polling
          pollTimerRef.current = setTimeout(
            () => void pollTicket(ticketId),
            POLL_INTERVAL,
          );
          return;
        }

        const data = (await res.json()) as {
          status: "pending" | "running" | "done" | "failed";
          progressStage?: string;
          results?: {
            jobs: Array<Record<string, unknown>>;
            queryStats: Array<{ query: string; resultCount: number }>;
          };
          error?: string;
        };

        if (data.progressStage) {
          setProgressStage(data.progressStage);
        }

        if (data.status === "done" && data.results) {
          // Terminal: success
          activeTicketRef.current = null;
          localStorage.removeItem("mkt_activeTicketId");
          if (scrapePhaseTimerRef.current) {
            clearInterval(scrapePhaseTimerRef.current);
            scrapePhaseTimerRef.current = null;
          }
          setScraping(false);
          const jobs = data.results.jobs ?? [];
          const stats = data.results.queryStats ?? [];
          setQueryStats(stats);
          const marketRaw = JSON.stringify(jobs);
          onScrapeComplete({
            marketRaw,
            queryStats: stats,
            marketSource: "api",
          });
        } else if (data.status === "failed") {
          // Terminal: failure
          activeTicketRef.current = null;
          localStorage.removeItem("mkt_activeTicketId");
          if (scrapePhaseTimerRef.current) {
            clearInterval(scrapePhaseTimerRef.current);
            scrapePhaseTimerRef.current = null;
          }
          setScraping(false);
          setScrapeError(data.error ?? "Scrape failed. Please try again.");
        } else {
          // Still running — poll again
          pollTimerRef.current = setTimeout(
            () => void pollTicket(ticketId),
            POLL_INTERVAL,
          );
        }
      } catch {
        // Network error — keep polling
        pollTimerRef.current = setTimeout(
          () => void pollTicket(ticketId),
          POLL_INTERVAL,
        );
      }
    },
    [onScrapeComplete],
  );

  // ── Query generation ──────────────────────────────────────────────────────

  const generateQueries = useCallback(async () => {
    if (!session.profile || !session.preferences || !session.context) {
      setGenError(
        "Profile, preferences, and context are required to generate queries.",
      );
      return;
    }

    setGeneratingQueries(true);
    setGenError(null);

    try {
      const avoidList = (session.profile.avoidTerms ?? [])
        .map((t) => (typeof t === "string" ? t : t.term))
        .filter(Boolean);

      const message = JSON.stringify({
        discipline: session.profile.discipline,
        seniority: session.profile.seniority,
        domains: session.profile.domains,
        hardSkills: session.profile.hardSkills?.slice(0, 15),
        targetDirection: session.profile.targetDirection,
        employmentTypes: session.preferences.employmentTypes,
        avoidTerms: avoidList,
        context: {
          careerDirection: session.context.careerDirection,
          careerDirectionText: session.context.careerDirectionText,
        },
      });

      const raw = await callClaude("QUERY_GEN", message, 512);
      const parsed = JSON.parse(cleanJson(raw)) as string[];

      if (!Array.isArray(parsed) || parsed.length === 0) {
        throw new Error("No queries generated");
      }

      const generated = parsed
        .filter((q) => typeof q === "string" && q.trim().length > 0)
        .slice(0, 6);

      setQueries(generated);
      onSessionUpdate({ generatedQueries: generated });
    } catch (e) {
      setGenError(
        e instanceof Error ? e.message : "Failed to generate queries",
      );
    } finally {
      setGeneratingQueries(false);
    }
  }, [session, onSessionUpdate]);

  // ── Query editing ──────────────────────────────────────────────────────────

  function startEdit(idx: number) {
    setEditingIdx(idx);
    setEditValue(queries[idx]);
  }

  function commitEdit(idx: number) {
    const trimmed = editValue.trim();
    if (!trimmed) {
      // Reject empty — restore original (GS-MKT-03)
      setEditingIdx(null);
      setEditValue("");
      return;
    }
    const updated = [...queries];
    updated[idx] = trimmed;
    setQueries(updated);
    onSessionUpdate({ generatedQueries: updated });
    setEditingIdx(null);
    setEditValue("");
  }

  function cancelEdit() {
    setEditingIdx(null);
    setEditValue("");
  }

  function removeQuery(idx: number) {
    // Must keep at least 1 query (GS-MKT-05)
    if (queries.length <= 1) return;
    const updated = queries.filter((_, i) => i !== idx);
    setQueries(updated);
    onSessionUpdate({ generatedQueries: updated });
  }

  function addQuery() {
    const trimmed = newQuery.trim();
    if (!trimmed) return;
    const updated = [...queries, trimmed];
    setQueries(updated);
    onSessionUpdate({ generatedQueries: updated });
    setNewQuery("");
  }

  // ── Avoid terms ───────────────────────────────────────────────────────────

  function removeAvoidTerm(idx: number) {
    const updated = avoidTerms.filter((_, i) => i !== idx);
    setAvoidTerms(updated);
    onSessionUpdate({
      profile: { ...session.profile!, avoidTerms: updated },
    });
  }

  // ── Manual URL (GS-MKT-21) ────────────────────────────────────────────────

  function handleAddManualUrl() {
    const trimmed = manualUrl.trim();
    if (!trimmed) return;
    if (!isLinkedInJobUrl(trimmed)) {
      setManualUrlError(
        "Please enter a valid LinkedIn job URL (linkedin.com/jobs/view/...)",
      );
      return;
    }
    setManualUrlError(null);
    setManualUrl("");
    // Append as a placeholder entry to marketRaw — would normally scrape the URL
    // Post-MVP: actual scraping. For now, acknowledge the intent.
    // This satisfies GS-MKT-21 acceptance criteria (input + validation)
  }

  // ── Scrape launch (GS-MKT-06, GS-MKT-29–39: v3 ticket model) ────────────

  async function launchScrape() {
    if (
      scraping ||
      queries.length === 0 ||
      (queueCooldownUntil !== null && Date.now() < queueCooldownUntil)
    )
      return;

    setScraping(true);
    setScrapeError(null);
    setScrapePhaseIdx(0);
    setProgressStage("queued");

    // Animate scrape phases
    let phaseIdx = 0;
    scrapePhaseTimerRef.current = setInterval(() => {
      phaseIdx = Math.min(phaseIdx + 1, SCRAPE_PHASES.length - 1);
      setScrapePhaseIdx(phaseIdx);
    }, 18000);

    try {
      const location =
        session.preferences?.preferredLocation ||
        session.preferences?.locationTypes?.join(", ") ||
        "United States";
      const employmentTypes = session.preferences?.employmentTypes ?? [
        "Full-time",
      ];
      const idempotencyKey = `mkt-${Date.now()}-${Math.random().toString(36).slice(2)}`;

      const res = await fetch(`${API_BASE}/jobs/scrape`, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "X-Idempotency-Key": idempotencyKey,
        },
        body: JSON.stringify({ queries, location, employmentTypes }),
      });

      if (!res.ok) {
        if (res.status === 503) {
          const cooldownEnd = Date.now() + 30_000;
          setQueueCooldownUntil(cooldownEnd);
          setScrapeError(
            "Market scraping temporarily unavailable — please try again in a minute.",
          );
          setScraping(false);
          if (scrapePhaseTimerRef.current)
            clearInterval(scrapePhaseTimerRef.current);
          // Auto-clear cooldown after 30s
          setTimeout(() => setQueueCooldownUntil(null), 30_000);
          return;
        }
        const errText = await res.text().catch(() => "");
        throw new Error(
          `Scrape request failed (${res.status}): ${errText.slice(0, 200)}`,
        );

interface MarketRawOutput {
  keywords?: Keyword[];
  compRanges?: { low: string; median: string; high: string } | null;
  jobTypes?: Array<{
    name: string;
    description: string;
    why: string;

function parseMarketOutput(raw: string): {
  marketAnalysis: MarketAnalysis;
  jobTypes: JobType[];
  miningQuestions: Array<{ id: string; question: string; why?: string }>;
} {
  const parsed = JSON.parse(cleanJson(raw)) as MarketRawOutput;

  const jobTypes: JobType[] = (parsed.jobTypes ?? []).map((jt) => ({
    name: jt.name ?? "",
    description: jt.description ?? "",
    why: jt.why ?? "",
    compRange: jt.compRange ?? null,
    volume: jt.volume ?? null,
    matchStrength: jt.matchStrength ?? "mid",
    selected: true, // GS-MKT-14: all default selected
    searchTerms: jt.searchTerms,
  }));

  const marketAnalysis: MarketAnalysis = {
    keywords: parsed.keywords ?? [],
    compRanges: parsed.compRanges ?? null,
    discoveryRecs: (parsed.discoveryRecs ?? []).map((r, i) => ({
      id: r.id ?? `rec-${i}`,
      name: r.name ?? "",
      rationale: r.rationale ?? "",
      compRange: r.compRange,
      volume: r.volume,
    })),
    exclusionTags: parsed.exclusionTags ?? [],
    educationVisibility: parsed.educationVisibility ?? "show",
  };

  return {
    marketAnalysis,
    jobTypes,
    miningQuestions: parsed.miningQuestions ?? [],
  };
}

// ── FOMO competitiveness delta ────────────────────────────────────────────────

function getFomoDelta(jobTypes: JobType[]): number {
  const allSelected = jobTypes.every((jt) => jt.selected);
  if (allSelected) return 0;
  const unselected = jobTypes.filter((jt) => !jt.selected);
  // Each unselected category represents a fraction of the category weight
  const perCat = WEIGHTS.categoriesSelected / Math.max(jobTypes.length, 1);
  return Math.round(unselected.length * perCat);
}

// ── Match strength badge label/color ─────────────────────────────────────────

function matchBadge(strength: "high" | "mid" | "low"): {
  label: string;
  color: string;
} {
  switch (strength) {
    case "high":
      return { label: "Top Candidate", color: "var(--primary, #ff5a17)" };
    case "mid":
      return { label: "Strong Fit", color: "var(--warning, #eab308)" };
    case "low":
      return { label: "Worth Exploring", color: "var(--text-muted)" };
  }
}

        const prepRaw = await callClaude("MARKET_PREP", payload, 4096);
        prepReport = prepRaw;

        // Save intermediate result (HYGIENE Rule 29)
        onSessionUpdate?.({ marketPrepReport: prepReport });
      } catch (e) {
        // GS-MKT-12: fallback — continue with raw data on MARKET_PREP failure
        if (isRocketError(e)) {
          setAnalysisError(
            `Insufficient rockets. ${e.remaining !== undefined ? `Balance: ${e.remaining}` : ""}`,
          );
          setIsRunning(false);
          onStatusChange?.("error", "Insufficient rockets");
          return;
        }
        // Non-rocket error: fall through to Phase 2 with raw data
        prepReport = "";
      }

      // ── Phase 2: MARKET ───────────────────────────────────────────────────

      try {
        // Build Phase 2 message
        let marketMessage: string;
        if (prepReport) {
          // Note: content derives from untrusted job data — wrapped in delimiter tags (RT-701)
          marketMessage = JSON.stringify({
            marketPrepReport: `<market_prep_output>\n${prepReport}\n</market_prep_output>`,
            profile: {
              discipline: profile?.discipline,
              seniority: profile?.seniority,
              hardSkills: profile?.hardSkills?.slice(0, 20),
              differentiators: profile?.differentiators,
              achievements: profile?.achievements?.slice(0, 5),
              targetDirection: profile?.targetDirection,
              avoidTerms: (profile?.avoidTerms ?? []).map((t) =>
                typeof t === "string" ? t : t.term,
              ),
            },
            employmentTypes: preferences?.employmentTypes ?? ["Full-time"],
          });
        } else {
          // Fallback: send raw data directly
          const { text } = preprocessMarketData(marketRaw);
          marketMessage = JSON.stringify({
            rawJobData: text,
            profile: {
              discipline: profile?.discipline,
              seniority: profile?.seniority,
              hardSkills: profile?.hardSkills?.slice(0, 20),
              differentiators: profile?.differentiators,
              achievements: profile?.achievements?.slice(0, 5),
              targetDirection: profile?.targetDirection,
              avoidTerms: (profile?.avoidTerms ?? []).map((t) =>
                typeof t === "string" ? t : t.term,
              ),
            },
            employmentTypes: preferences?.employmentTypes ?? ["Full-time"],
          });
        }

        const marketRaw2 = await callClaude("MARKET", marketMessage, 4096);
        const {
          marketAnalysis: ma,
          jobTypes: jts,
          miningQuestions,
        } = parseMarketOutput(marketRaw2);

        setMarketAnalysis(ma);
        setJobTypes(jts);

        // Save analysis results (GS-MKT-13, HYGIENE Rule 29)
        // Note: jobTypes NOT saved here — only via explicit Lock (GS-MKT-20 note)
        onSessionUpdate?.({
          marketPrepReport: prepReport || undefined,
          marketAnalysis: ma,
          miningQuestions,
        });

        setViewPhase("categories");
        setIsRunning(false);
        onStatusChange?.("done", undefined, jobCount);
      } catch (e) {
        const msg = isRocketError(e)
          ? `Insufficient rockets for analysis.`
          : e instanceof Error
            ? e.message
            : "Market analysis failed";
        setAnalysisError(msg);
        setIsRunning(false);
        onStatusChange?.("error", msg, jobCount);
      }
    },
    [session, onStatusChange, onSessionUpdate],
  );

  // ── Auto-run on mount ─────────────────────────────────────────────────────

  useEffect(() => {
    hasRun.current = false; // BUG-035: reset so React Strict Mode remount doesn't skip analysis
    if (hasRun.current) return;

// ── Substep architecture ──────────────────────────────────────────────────────
// substep 0: StepCollect — query editor + scrape launch + scrape progress
// substep 1: StepCollect (analysis-running mode) + Step6Analysis hidden
// substep 2: Step6Analysis (categories view)

type Substep = 0 | 1 | 2;

export function AimPage() {
  const [session, setSession] = useState<SessionData | null>(null);
  const [loading, setLoading] = useState(true);
  const [substep, setSubstep] = useState<Substep>(0);

  // Analysis status reported by Step6Analysis to show in StepCollect
  const [analysisStatus, setAnalysisStatus] = useState<
    "idle" | "running" | "done" | "error"
  >("idle");
  const [analysisError, setAnalysisError] = useState<string | undefined>(
    undefined,
  );
  const [analysisJobCount, setAnalysisJobCount] = useState<number | undefined>(
    undefined,
  );

  const saveInProgress = useRef(false);

  // ── Load session on mount ─────────────────────────────────────────────────

  useEffect(() => {
    loadSession()
      .then((s) => {
        const loaded = s ?? ({} as SessionData);
        setSession(loaded);

        // Determine initial substep from session state
        if (loaded.marketAnalysis && loaded.jobTypes?.length) {
          // Analysis already done — show categories
          setSubstep(2);
          setAnalysisStatus("done");
        } else if (loaded.marketRaw) {
          // Scrape done, analysis not yet complete — trigger analysis
          setSubstep(1);
        } else {
          // Fresh entry — query editor
          setSubstep(0);
        }
      })
      .catch(() => {
        setSession({} as SessionData);
        setSubstep(0);
      })
      .finally(() => setLoading(false));
  }, []);

  // ── Session update helper (BUG-012: always read fresh) ───────────────────

  const handleSessionUpdate = useCallback(
    async (data: Partial<SessionData>) => {
      if (saveInProgress.current) return;
      saveInProgress.current = true;
      try {
        // BUG-012: fresh loadSession each time
        const latest = (await loadSession()) ?? ({} as SessionData);
        const updated = { ...latest, ...data };
        await saveSession(updated);
        setSession(updated);
      } finally {
        saveInProgress.current = false;
      }
    },
    [],
  );

  // ── Scrape complete callback ───────────────────────────────────────────────
  // Called by StepCollect when BD polling finishes.
  // HYGIENE Rule 29: saveSession BEFORE advancing substep.

  const handleScrapeComplete = useCallback(
    async (data: {
      marketRaw: string;
      queryStats: Array<{ query: string; resultCount: number }>;
      marketSource: "api";
    }) => {
      // Save scrape results before advancing
      const latest = (await loadSession()) ?? ({} as SessionData);
      const updated: SessionData = {
        ...latest,
        marketRaw: data.marketRaw,
        queryStats: data.queryStats,
        marketSource: data.marketSource,
      };
      await saveSession(updated);
      setSession(updated);

      // HYGIENE Rule 29: save complete, now advance substep
      setSubstep(1);
      setAnalysisStatus("idle");
    },
    [],
  );

  // ── Analysis status callback ───────────────────────────────────────────────
  // Step6Analysis reports status so StepCollect can render the progress overlay.

      status: "idle" | "running" | "done" | "error",
      error?: string,
      jobCount?: number,
    ) => {
      setAnalysisStatus(status);
      setAnalysisError(error);
      setAnalysisJobCount(jobCount);
    },
    [],
  );

  // ── Analysis session update ───────────────────────────────────────────────
  // Step6Analysis saves intermediate results (marketPrepReport, marketAnalysis).
  // NOTE: does NOT include jobTypes (GS-MKT-20).

  const handleAnalysisSessionUpdate = useCallback(
    async (data: Partial<SessionData>) => {
      await handleSessionUpdate(data);
    },
    [handleSessionUpdate],
  );

  // ── View targets callback ─────────────────────────────────────────────────
  // When analysis is done and user taps "View Your Targets", advance substep.
  // HYGIENE Rule 29: session already saved by analysis pipeline; just advance.

  const handleViewTargets = useCallback(() => {
    setSubstep(2);
  }, []);

  // ── Lock categories (complete) ────────────────────────────────────────────
  // Step6Analysis calls onComplete with jobTypes + rankedCategories + currentStep.
  // HYGIENE Rule 29: save BEFORE state update.
  // GS-MKT-20: hardcoded currentStep: 6 (HYGIENE Rule 56).
  // BUG-074: never write currentStep > maxStep+1.

  const handleComplete = useCallback(async (data: Partial<SessionData>) => {
    const latest = (await loadSession()) ?? ({} as SessionData);
    const currentMax = latest.maxStep ?? latest.currentStep ?? 0;
    const nextStep = data.currentStep ?? 6;
    // BUG-074: clamp advancement
    const safeStep = Math.min(nextStep, currentMax + 1);

    const updated: SessionData = {
      ...latest,
      ...data,
      currentStep: safeStep,
      maxStep: Math.max(currentMax, safeStep),
    };

    // HYGIENE Rule 29: save before page re-init
    await saveSession(updated);
    setSession(updated);

    // Trigger page.tsx re-render at new step (self-managing page pattern)
    window.location.reload();
  }, []);

  // ── Render ────────────────────────────────────────────────────────────────

  if (loading || !session) {
    return (
      <div
        style={{
          display: "flex",
          alignItems: "center",
          justifyContent: "center",
          minHeight: "60vh",
          color: "var(--text-muted)",
          fontSize: "14px",
        }}
      >
        Loading...
      </div>
    );
  }

  const containerStyle: React.CSSProperties = {
    minHeight: "100vh",
    background: "var(--background)",
    color: "var(--text)",
    fontFamily: "var(--font-body)",
  };

  // Step6Analysis is mounted hidden in substeps 1-2 so it can run the pipeline
  // while StepCollect shows the progress overlay (PRD Section 8 implementation).
  const step6Visible = substep === 2;

  return (
    <div style={containerStyle}>
      {/* StepCollect — visible in substeps 0 and 1 */}
      {substep <= 1 && (
        <StepCollect
          session={session}
          substep={substep}
          analysisStatus={substep === 1 ? analysisStatus : "idle"}
          analysisError={substep === 1 ? analysisError : undefined}
          jobCount={analysisJobCount}
          onScrapeComplete={handleScrapeComplete}
          onViewTargets={handleViewTargets}
          onSessionUpdate={(data) => void handleSessionUpdate(data)}
        />
      )}

      {/* Step6Analysis — mounted (hidden) in substep 1 to run pipeline,
          fully visible in substep 2 (categories view) */}
      {substep >= 1 && (
        <div style={{ display: step6Visible ? "block" : "none" }}>
          <Step6Analysis
            session={session}
            initialViewPhase={substep === 1 ? "analyzing" : "categories"}
            onStatusChange={handleAnalysisStatusChange}
            onComplete={(data) => void handleComplete(data)}
            onSessionUpdate={(data) => void handleAnalysisSessionUpdate(data)}
          />
        </div>
      )}
    </div>
  );
}

// ── Daily budget tracking ─────────────────────────────────────────────────────

const DAILY_BD_BUDGET = 100;

function dailyBudgetKey(): string {
  const day = new Date().toISOString().slice(0, 10);
  return `bd:budget:${day}`;
}

function getRedis(): Redis | null {
  if (
    !process.env.UPSTASH_REDIS_REST_URL ||
    !process.env.UPSTASH_REDIS_REST_TOKEN
  ) {
    return null;
  }
  return new Redis({
    url: process.env.UPSTASH_REDIS_REST_URL,
    token: process.env.UPSTASH_REDIS_REST_TOKEN,
  });
}

async function checkAndIncrementBudget(): Promise<{
  allowed: boolean;
  used: number;
}> {
  const redis = getRedis();
  if (!redis) {
    // No Redis configured — allow (dev mode)
    return { allowed: true, used: 0 };
  }
  const key = dailyBudgetKey();
  const current = (await redis.get<number>(key)) ?? 0;
  if (current >= DAILY_BD_BUDGET) {
    return { allowed: false, used: current };
  }
  await redis.incr(key);
  // Expire at end of day
  await redis.expire(key, 86400);
  return { allowed: true, used: current + 1 };
}

// ── Rate limiting (10 req/min per IP) ────────────────────────────────────────

function getRatelimit(): Ratelimit | null {
  const redis = getRedis();
  if (!redis) return null;
  return new Ratelimit({
    redis,
    limiter: Ratelimit.slidingWindow(10, "1 m"),
    prefix: "bd:rl",
  });
}

interface BDTriggerResponse {

interface BDProgressResponse {
  snapshot_id: string;
  status: "starting" | "running" | "ready" | "failed";
}

/**
 * Trigger one BD discovery snapshot for a single keyword+employmentType combo.
 * BUG-031: BD uses keyword input, NOT LinkedIn search URLs.
 */
async function triggerSnapshot(
  keyword: string,
  location: string,
  employmentType: string,
  apiKey: string,
): Promise<string> {
  const url = `${BD_API_BASE}/datasets/v3/trigger?dataset_id=${BD_DATASET_ID}&type=discover_new&discover_by=keyword&limit_per_input=30&format=json&include_errors=true`;

  const body = JSON.stringify([
    {
      keyword,
      location: location || "United States",
      country: "US",
      job_type: employmentType,
    },
  ]);

  const res = await fetch(url, {
    method: "POST",
    headers: {
      Authorization: `Bearer ${apiKey}`,
      "Content-Type": "application/json",
    },
    body,
  });

  if (!res.ok) {
    const text = await res.text().catch(() => "");
    throw new Error(`BD trigger failed (${res.status}): ${text.slice(0, 200)}`);
  }

  const data = (await res.json()) as BDTriggerResponse;
  return data.snapshot_id;
}

/**
 * Check snapshot status.
 */
async function checkSnapshot(
  snapshotId: string,
  apiKey: string,
): Promise<BDProgressResponse> {
  const url = `${BD_API_BASE}/datasets/v3/snapshot/${snapshotId}/progress`;
  const res = await fetch(url, {
    headers: { Authorization: `Bearer ${apiKey}` },
  });
  if (!res.ok) {
    throw new Error(`BD progress check failed (${res.status})`);
  }
  return (await res.json()) as BDProgressResponse;
}

/**
 * Download snapshot data.
 */
async function downloadSnapshot(
  snapshotId: string,
  apiKey: string,

  if (!res.ok) {

  }
  try {

    return Array.isArray(data) ? data : [];
  } catch {
    return [];
  }
}

/**
 * Deduplicate jobs by title+company (case-insensitive).
 * Also tracks per-query result counts (before dedup).
 */
function deduplicateJobs(
  jobsByQuery: Array<{ query: string; jobs: JobListing[] }>,
): { jobs: JobListing[]; queryStats: QueryStat[] } {
  const seen = new Set<string>();
  const deduped: JobListing[] = [];
  const queryStats: QueryStat[] = [];

  for (const { query, jobs } of jobsByQuery) {
    queryStats.push({ query, resultCount: jobs.length });
    for (const job of jobs) {
      const key = `${job.title.toLowerCase().trim()}|${job.company.toLowerCase().trim()}`;
      if (!seen.has(key) && (job.title || job.company)) {
        seen.add(key);
        deduped.push(job);
      }
    }
  }

  return { jobs: deduped, queryStats };
}

// ── Cycle employment types across queries ─────────────────────────────────────

function cycleEmploymentTypes(
  queries: string[],
  employmentTypes: string[],
): Array<{ query: string; employmentType: string }> {
  if (!employmentTypes || employmentTypes.length === 0) {
    return queries.map((q) => ({ query: q, employmentType: "Full-time" }));
  }
  const mapped = employmentTypes.map(
    (et) => EMPLOYMENT_TYPE_MAP[et] ?? "Full-time",
  );
  return queries.map((q, i) => ({
    query: q,
    employmentType: mapped[i % mapped.length],
  }));
}

// ── POST handler ──────────────────────────────────────────────────────────────

export async function POST(req: NextRequest) {
  // HYGIENE Rule 27: validateOrigin as boolean if-check, never try/catch
  if (!validateOrigin(req)) {
    return NextResponse.json({ error: "Forbidden" }, { status: 403 });
  }

  // Rate limit: 10 requests per minute per IP (GS-MKT-10)
  const ip =
    req.headers.get("x-forwarded-for")?.split(",")[0]?.trim() ?? "unknown";
  const ratelimit = getRatelimit();
  if (ratelimit) {
    const { success, reset } = await ratelimit.limit(ip);
    if (!success) {
      return NextResponse.json(
        { error: "Too many requests. Please wait and try again." },
        {
          status: 429,
          headers: {
            "Retry-After": String(Math.ceil((reset - Date.now()) / 1000)),
          },
        },
      );
    }
  }

  const apiKey = process.env.BRIGHTDATA_API_KEY;

  let body: Record<string, unknown>;
  try {
    body = (await req.json()) as Record<string, unknown>;
  } catch {
    return NextResponse.json(
      { error: "Invalid request body" },
      { status: 400 },
    );
  }

  const action = body.action as string;

  // ── TRIGGER ──────────────────────────────────────────────────────────────

  if (action === "trigger") {
    // RT-201: auth guard — reject unauthenticated trigger requests
    const authToken = await getAuthToken();
    if (!authToken || !verifyJWT(authToken)) {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }

    // Check daily budget (GS-MKT-10)
    const { allowed, used } = await checkAndIncrementBudget();
    if (!allowed) {
      return NextResponse.json(
        {
          error: `Daily scrape budget exhausted (${used}/${DAILY_BD_BUDGET} requests used today). Try again tomorrow.`,
        },
        { status: 429 },
      );
    }

    const queries = body.queries as string[] | undefined;
    const location = (body.location as string) || "United States";
    const employmentTypes = (body.employmentTypes as string[]) || ["Full-time"];

    if (!queries || !Array.isArray(queries) || queries.length === 0) {
      return NextResponse.json(
        { error: "At least one search query is required" },
        { status: 400 },
      );
    }

    // Dev/test mode — no real API key
    if (!apiKey) {
      // Return mock snapshot IDs for local development
      const snapshotIds = queries.map((_, i) => `mock-snapshot-${i}`);
      const queryMap: Record<string, string> = {};
      queries.forEach((q, i) => {
        queryMap[snapshotIds[i]] = q;
      });
      return NextResponse.json({ snapshotIds, queryMap });
    }

    const queryEmploymentPairs = cycleEmploymentTypes(queries, employmentTypes);

    const snapshotIds: string[] = [];
    const queryMap: Record<string, string> = {};
    const errors: string[] = [];

    // Trigger one snapshot per query (parallel)
    await Promise.allSettled(
      queryEmploymentPairs.map(async ({ query, employmentType }) => {
        try {
          const snapshotId = await triggerSnapshot(
            query,
            location,
            employmentType,
            apiKey,
          );
          snapshotIds.push(snapshotId);
          queryMap[snapshotId] = query;
        } catch (e) {
          errors.push(
            e instanceof Error ? e.message : `Trigger failed for: ${query}`,
          );
        }
      }),
    );

    if (snapshotIds.length === 0) {
      return NextResponse.json(
        { error: `All scrape triggers failed: ${errors.join("; ")}` },
        { status: 502 },
      );
    }

    return NextResponse.json({
      snapshotIds,
      queryMap,
      warnings: errors.length > 0 ? errors : undefined,
    });
  }

  // ── POLL ──────────────────────────────────────────────────────────────────

  if (action === "poll") {
    const snapshotIds = body.snapshotIds as string[] | undefined;
    const queryMap = (body.queryMap as Record<string, string>) || {};
    const force = body.force === true;

    if (
      !snapshotIds ||
      !Array.isArray(snapshotIds) ||
      snapshotIds.length === 0
    ) {
      return NextResponse.json(
        { error: "snapshotIds required for poll" },
        { status: 400 },
      );
    }

    // RT-202: validate snapshot ID format before using in URL construction
    const snapshotIdRegex = /^[a-zA-Z0-9_-]{8,64}$/;
    for (const sid of snapshotIds) {
      if (!snapshotIdRegex.test(sid)) {
        return NextResponse.json(
          { error: "Invalid snapshot ID format" },
          { status: 400 },
        );
      }
    }

    // Dev mock mode
    if (!apiKey) {
      // Return mock results
      const mockJobs: JobListing[] = snapshotIds.flatMap((sid) => {
        const query = queryMap[sid] ?? "mock query";
        return [
          normalizeJob(
            {
              title: `Mock Job for ${query}`,
              company: "Mock Corp",
              location: "Remote",
              description: "Mock job description for development testing.",
              employment_type: "Full-time",
            },
            query,
          ),
        ];
      });
      const queryStats = snapshotIds.map((sid) => ({
        query: queryMap[sid] ?? sid,
        resultCount: 1,
      }));
      return NextResponse.json({
        done: true,
        jobs: mockJobs,
        total: mockJobs.length,
        pending: 0,
        queryStats,
      });
    }

    // Check progress for all snapshots in parallel
    const progressResults = await Promise.allSettled(
      snapshotIds.map((sid) => checkSnapshot(sid, apiKey)),
    );

    const readyIds: string[] = [];
    const pendingIds: string[] = [];

    progressResults.forEach((result, i) => {
      const sid = snapshotIds[i];
      if (result.status === "fulfilled") {
        const { status } = result.value;
        if (status === "ready") {
          readyIds.push(sid);
        } else if (status === "failed") {
          // Treat failed snapshots as done (skip them)
        } else {
          pendingIds.push(sid);
        }
      } else {
        // Error checking progress — treat as pending
        pendingIds.push(sid);
      }
    });

    const allDone = pendingIds.length === 0;
    const shouldReturn = allDone || force;

    if (!shouldReturn) {
      return NextResponse.json({
        done: false,
        pending: pendingIds.length,
        total: snapshotIds.length,
      });
    }

    // Download ready snapshots
    const jobsByQuery: Array<{ query: string; jobs: JobListing[] }> = [];

    await Promise.allSettled(
      readyIds.map(async (sid) => {
        const query = queryMap[sid] ?? sid;
        try {
          const rawJobs = await downloadSnapshot(sid, apiKey);
          const normalized = rawJobs.map((r) => normalizeJob(r, query));
          jobsByQuery.push({ query, jobs: normalized });
        } catch {
          jobsByQuery.push({ query, jobs: [] });
        }
      }),
    );

    // Add zero-result entries for snapshots that weren't ready (forced partial)
    for (const sid of pendingIds) {
      const query = queryMap[sid] ?? sid;
      const alreadyHasEntry = jobsByQuery.some((jbq) => jbq.query === query);
      if (!alreadyHasEntry) {
        jobsByQuery.push({ query, jobs: [] });
      }
    }

    const { jobs, queryStats } = deduplicateJobs(jobsByQuery);

    return NextResponse.json({
      done: true,
      jobs,
      total: jobs.length,
      pending: pendingIds.length,
      queryStats,
    });
  }

  return NextResponse.json(
    { error: `Unknown action: ${action}` },
    { status: 400 },
  );
}
