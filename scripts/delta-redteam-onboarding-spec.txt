export interface ParseBannerProps {

  onCancel?: () => void;

export function ParseBanner({ status, onRetry, onCancel }: ParseBannerProps) {
  const cfg = {
    parsing: {
      bg: "var(--warning-light)",
      border: "var(--warning-border)",
      color: "var(--warning)",
      text: "Parsing your resume in the background...",
    },
    done: {
      bg: "var(--success-light)",
      border: "var(--success-border)",
      color: "var(--success)",
      text: "Resume parsed ✓",
    },
    error: {
      bg: "var(--error-light)",
      border: "var(--error-border)",
      color: "var(--error)",
      text: "Resume parsing failed. Retry?",
    },
    cancelled: {
      bg: "var(--surface)",
      border: "var(--border)",
      color: "var(--text-muted)",
      text: "Cancelled. You can try again.",
    },
  }[status];

  return (
    <div
      role="status"
      aria-live="polite"
      style={{
        background: cfg.bg,
        border: `1px solid ${cfg.border}`,
        borderRadius: "var(--radius)",
        padding: "8px 12px",
        display: "flex",
        alignItems: "center",
        justifyContent: "space-between",
        gap: 8,
        flexWrap: "wrap",
      }}
    >
      <span style={{ fontSize: 13, color: cfg.color }}>{cfg.text}</span>
      <div style={{ display: "flex", gap: 6 }}>
        {status === "error" && onRetry && (
          <Btn
            variant="ghost"
            size="sm"
            onClick={onRetry}
            aria-label="Retry resume parsing"
          >
            Retry
          </Btn>
        )}
        {status === "parsing" && onCancel && (
          <Btn
            variant="ghost"
            size="sm"
            onClick={onCancel}
            aria-label="Cancel resume parsing"
          >
            Cancel
          </Btn>
        )}
      </div>
    </div>
  );
}

export function Step1Resume({
  onUpload,
  parseStatus = "idle",
  onRetryParse,
  onCancelParse,
  initialResumeRaw,
}: Step1ResumeProps) {
  const [tab, setTab] = useState<Tab>("file");
  const [isDragOver, setIsDragOver] = useState(false);
  const [procState, setProcState] = useState<ProcessingState>("idle");
  const [fileError, setFileError] = useState<string | null>(null);
  const [pasteText, setPasteText] = useState("");
  const [pasteError, setPasteError] = useState<string | null>(null);
  // BUG-017: show "continue" when returning user has existing resume
  const [showContinue, setShowContinue] = useState(!!initialResumeRaw);
  const fileInputRef = useRef<HTMLInputElement>(null);
  const successTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  useEffect(() => {
    return () => {
      if (successTimerRef.current) clearTimeout(successTimerRef.current);
    };
  }, []);

  // Keep continue state in sync if initialResumeRaw changes
  useEffect(() => {
    if (initialResumeRaw) setShowContinue(true);
  }, [initialResumeRaw]);

  async function processFile(file: File) {
    setFileError(null);
    const check = validateFile(file);
    if (!check.valid) {
      setFileError(check.error);
      return;
    }
    setProcState("processing");
    try {
      const text = await extractText(file);
      if (!text || text.trim().length === 0) {
        setProcState("error");
        setFileError("No text could be extracted from this file.");
        return;
      }
      setProcState("success");
      // Brief success animation (~1s) before advancing (INPUTS.md)
      successTimerRef.current = setTimeout(() => {
        onUpload(text);
      }, 900);
    } catch (err) {
      setProcState("error");
      setFileError(
        err instanceof Error ? err.message : "Failed to process file.",
      );
    }
  }

  function handleDrop(e: React.DragEvent) {
    e.preventDefault();
    setIsDragOver(false);
    const file = e.dataTransfer.files[0];
    if (file) processFile(file);
  }

  function handleFileChange(e: React.ChangeEvent<HTMLInputElement>) {
    const file = e.target.files?.[0];
    if (file) processFile(file);
    e.target.value = "";
  }

  function handlePasteSubmit() {
    setPasteError(null);
    const trimmed = pasteText.trim();
    if (!trimmed) {
      setPasteError("Please paste your resume text first.");
      return;
    }
    if (trimmed.length < 50) {
      setPasteError("Text is too short. Please paste your full resume.");
      return;
    }
    onUpload(trimmed);
  }

  function handleUploadDifferent() {
    setShowContinue(false);
    setProcState("idle");
    setFileError(null);
    setPasteText("");
    setPasteError(null);
  }

  // ── Returning user: show Continue + Upload different ──────────────────────
  // BUG-017: re-entry flow
  if (showContinue && procState === "idle") {
    return (
      <div style={outerStyle}>
        <div style={innerStyle}>
          <div>
            <h2 style={h2Style}>Upload your resume</h2>
            <p style={subtitleStyle}>
              We extracted this from your resume. Edit anything that looks off.
            </p>
          </div>

          {parseStatus !== "idle" && (
            <ParseBanner
              status={parseStatus}
              onRetry={onRetryParse}
              onCancel={onCancelParse}
            />
          )}

          <div style={{ display: "flex", flexDirection: "column", gap: 10 }}>
            <Btn
              variant="primary"
              size="lg"
              onClick={() => initialResumeRaw && onUpload(initialResumeRaw)}
              style={{ width: "100%", justifyContent: "center" }}
              aria-label="Continue with existing resume"
            >
              Continue with existing resume
            </Btn>
            <Btn
              variant="ghost"
              size="md"
              onClick={handleUploadDifferent}
              style={{ width: "100%", justifyContent: "center" }}
              aria-label="Upload a different resume"
            >
              Upload different
            </Btn>
          </div>
        </div>
      </div>
    );
  }

  // ── Processing state ───────────────────────────────────────────────────────
  if (procState === "processing") {
    return (
      <div style={outerStyle}>
        <div style={{ ...innerStyle, alignItems: "center", gap: 20 }}>
          <Spin size="lg" />
          <div style={{ textAlign: "center" }}>
            <p
              style={{
                fontSize: 16,
                fontWeight: 600,
                color: "var(--text)",
                margin: "0 0 12px 0",
              }}
            >
              Processing your resume...
            </p>
            <div style={{ display: "flex", flexDirection: "column", gap: 4 }}>
              {[
                "Reading document…",
                "Extracting structure…",
                "Building profile…",
              ].map((phase) => (
                <p
                  key={phase}
                  style={{
                    fontSize: 13,
                    color: "var(--text-muted)",
                    margin: 0,
                  }}
                >
                  {phase}
                </p>
              ))}
            </div>
          </div>
        </div>
      </div>
    );
  }

  // ── Success state ──────────────────────────────────────────────────────────
  if (procState === "success") {
    return (
      <div style={outerStyle}>
        <div style={{ ...innerStyle, alignItems: "center", gap: 16 }}>
          <div
            style={{
              width: 52,
              height: 52,
              borderRadius: "50%",
              background: "var(--success-light)",
              border: "2px solid var(--success-border)",
              display: "flex",
              alignItems: "center",
              justifyContent: "center",
              fontSize: 24,
              color: "var(--success)",
            }}
            aria-hidden="true"
          >
            ✓
          </div>
          <p
            style={{
              fontSize: 15,
              fontWeight: 600,
              color: "var(--text)",
              textAlign: "center",
              margin: 0,
            }}
          >
            Got it! Setting things up...
          </p>
        </div>
      </div>
    );
  }

  // ── Normal upload UI ───────────────────────────────────────────────────────
  return (
    <div style={outerStyle}>
      <div style={innerStyle}>
        {/* Heading */}
        <div>
          <h2 style={h2Style}>Upload your resume</h2>
          <p style={subtitleStyle}>
            We&apos;ll extract your info and get everything set up fast.
          </p>
        </div>

        {/* Trust signals */}
        <div style={{ display: "flex", gap: 8, flexWrap: "wrap" }}>
          {[
            "Processed locally first",
            "Encrypted in your browser",
            "No data stored by AI",
          ].map((t) => (
            <span
              key={t}
              style={{
                fontSize: 11,
                color: "var(--text-muted)",
                background: "var(--surface)",
                border: "1px solid var(--border)",
                borderRadius: "var(--radius-full)",
                padding: "3px 10px",
              }}
            >
              {t}
            </span>
          ))}
        </div>

        {/* Parse status banner (shown when parseStatus is active) */}
        {parseStatus !== "idle" && (
          <ParseBanner
            status={parseStatus}
            onRetry={onRetryParse}
            onCancel={onCancelParse}
          />
        )}

        {/* Tabs */}
        <div
          style={{
            display: "flex",
            borderBottom: "1px solid var(--border)",
          }}
          role="tablist"
        >
          {(["file", "paste"] as Tab[]).map((t) => (
            <button
              key={t}
              role="tab"
              aria-selected={tab === t}
              onClick={() => setTab(t)}
              style={{
                background: "transparent",
                border: "none",
                padding: "8px 16px",
                fontSize: 13,
                fontFamily: "var(--font-body)",
                color: tab === t ? "var(--primary)" : "var(--text-muted)",
                cursor: "pointer",
                borderBottom: `2px solid ${tab === t ? "var(--primary)" : "transparent"}`,
                marginBottom: "-1px",
                transition: "color 0.15s",
                fontWeight: tab === t ? 600 : 400,
              }}
            >
              {t === "file" ? "Upload file" : "Paste text"}
            </button>
          ))}
        </div>

        {/* Tab content */}
        {tab === "file" ? (
          <div role="tabpanel">
            {/* Drop zone */}
            <div
              onDrop={handleDrop}
              onDragOver={(e) => {
                e.preventDefault();
                setIsDragOver(true);
              }}
              onDragLeave={() => setIsDragOver(false)}
              onClick={() => fileInputRef.current?.click()}
              onKeyDown={(e) =>
                (e.key === "Enter" || e.key === " ") &&
                fileInputRef.current?.click()
              }
              role="button"
              tabIndex={0}
              aria-label="Drop your resume here or click to browse"
              style={{
                border: `2px dashed ${isDragOver ? "var(--primary)" : "var(--border)"}`,

                background: isDragOver
                  ? "rgba(255,90,23,0.04)"

                style={{
                  fontSize: 15,

                  margin: "0 0 4px 0",

            role="tabpanel"
            style={{ display: "flex", flexDirection: "column", gap: 10 }}

            <textarea

                setPasteText(e.target.value);
                if (pasteError) setPasteError(null);

// ── Direction sub-step ────────────────────────────────────────────────────────

function DirectionStep({
  ctx,
  onNext,
  onBack,
}: {
  ctx: Partial<Context>;
  onNext: (ctx: Partial<Context>) => void;
  onBack?: () => void;
}) {
  const [selected, setSelected] = useState<string[]>(
    ctx.directionPresets?.length ? ctx.directionPresets : ["same"],
  );
  const [detail, setDetail] = useState(ctx.careerDirectionText ?? "");
  const [detailError, setDetailError] = useState("");

  const needsDetail = selected.some(
    (id) => DIRECTION_PRESETS.find((p) => p.id === id)?.needsDetail,
  );

  function togglePreset(id: string) {
    setSelected((prev) =>
      prev.includes(id) ? prev.filter((x) => x !== id) : [...prev, id],
    );
    setDetailError("");
  }

  function handleNext() {
    if (!selected.length) return;
    if (needsDetail && detail.trim().length < 10) {
      setDetailError(
        "Please describe your direction (at least 10 characters).",
      );
      return;
    }
    onNext({
      directionPresets: selected,
      careerDirectionText: needsDetail ? detail.trim() : detail,
      careerDirection: needsDetail ? detail.trim() : "",
    });
  }

  return (
    <div style={cardStyle}>
      <div>
        <h2 style={h2Style}>What are you looking for?</h2>
        <p style={subtitleStyle}>Pick the direction that fits best.</p>
      </div>

      <div style={{ display: "flex", flexDirection: "column", gap: 8 }}>
        {DIRECTION_PRESETS.map((preset) => (
          <div
            key={preset.id}
            role="checkbox"
            aria-checked={selected.includes(preset.id)}
            tabIndex={0}
            onClick={() => togglePreset(preset.id)}
            onKeyDown={(e) =>
              (e.key === "Enter" || e.key === " ") && togglePreset(preset.id)
            }
            style={{
              padding: "14px 16px",
              borderRadius: "var(--radius)",
              border: `1px solid ${selected.includes(preset.id) ? "var(--primary)" : "var(--border)"}`,
              background: selected.includes(preset.id)
                ? "rgba(255,90,23,0.08)"
                : "var(--surface)",
              cursor: "pointer",
              fontSize: 14,
              fontWeight: selected.includes(preset.id) ? 600 : 400,
              color: "var(--text)",
              transition: "all 0.15s",
            }}
          >
            {preset.label}

// ── Compensation sub-step ─────────────────────────────────────────────────────

function CompStep({
  prefs,
  onNext,
  onBack,
}: {
  prefs: Partial<Preferences>;
  onNext: (prefs: Partial<Preferences>) => void;
  onBack: () => void;
}) {
  const [compStructure, setCompStructure] = useState(
    prefs.compStructure ?? "Open to discussion",
  );
  const [hourly, setHourly] = useState(prefs.hourlyFloor ?? "");
  const [salary, setSalary] = useState(prefs.salaryFloor ?? "");
  const [errors, setErrors] = useState<{ hourly?: string; salary?: string }>(
    {},
  );

  const empTypes = prefs.employmentTypes ?? ["Full-time"];
  const hasContract =
    empTypes.includes("Contract") || empTypes.includes("Part-time");
  const hasFT = empTypes.includes("Full-time");
  const showBoth = hasContract && hasFT;
  const showHourlyPrimary = hasContract && !hasFT;

  // Bidirectional sync: hourly × 2080 = annual
  function onHourlyChange(val: string) {
    setHourly(val);
    setErrors((e) => ({ ...e, hourly: undefined }));
    const n = parseFloat(val);
    if (!isNaN(n) && n > 0) {
      setSalary(String(Math.round(n * 2080)));
    } else if (val === "") {
      setSalary("");
    }
  }

  function onSalaryChange(val: string) {
    setSalary(val);
    setErrors((e) => ({ ...e, salary: undefined }));
    const n = parseFloat(val.replace(/[,$]/g, ""));
    if (!isNaN(n) && n > 0) {
      setHourly(String(Math.round((n / 2080) * 100) / 100));
    } else if (val === "") {
      setHourly("");
    }
  }

  function validate(): boolean {
    const newErrors: { hourly?: string; salary?: string } = {};
    // BUG-019: zero/negative compensation blocked
    if (hourly !== "") {
      const n = parseFloat(hourly);
      if (isNaN(n) || n <= 0) {
        newErrors.hourly = "Hourly rate must be a positive number.";
      }
    }
    if (salary !== "") {
      const n = parseFloat(salary.replace(/[,$]/g, ""));
      if (isNaN(n) || n <= 0) {
        newErrors.salary = "Annual salary must be a positive number.";
      }
    }
    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  }

  function handleNext() {
    if (!validate()) return;
    onNext({
      compStructure,
      hourlyFloor: hourly,
      salaryFloor: salary,
    });
  }

  return (
    <div style={cardStyle}>
      <div>
        <h2 style={h2Style}>What are your compensation expectations?</h2>
      </div>

      <Sel
        label="Compensation structure"
        value={compStructure}
        onChange={setCompStructure}
        options={COMP_STRUCTURES.map((s) => ({ value: s, label: s }))}
      />

      {/* Adaptive compensation inputs */}
      {(showHourlyPrimary || showBoth) && (
        <Inp
          label={showBoth ? "Hourly floor" : "Hourly floor"}
          value={hourly}
          onChange={onHourlyChange}
          placeholder="e.g. 75"
          type="number"
          error={errors.hourly}
        />
      )}

      {(hasFT || showBoth) && (
        <Inp
          label={showBoth ? "Annual salary floor" : "Annual salary floor"}
          value={salary}
          onChange={onSalaryChange}
          placeholder="e.g. $120,000"
          error={errors.salary}
        />
      )}

      {/* Bidirectional sync helper */}
      {hourly && salary && (
        <p style={{ fontSize: 12, color: "var(--text-muted)", margin: 0 }}>
          ${hourly}/hr ↔ ${parseInt(salary).toLocaleString()}/yr (based on 2,080
          hrs/yr)
        </p>
      )}

      <div style={navRowStyle}>
        <Btn variant="ghost" onClick={onBack} aria-label="Go back">
          ← Back
        </Btn>
        <Btn
          variant="primary"
          onClick={handleNext}
          aria-label="Continue to location"
        >
          Next →
        </Btn>
      </div>
    </div>
  );
}

// ── Step3Preferences (orchestrator) ──────────────────────────────────────────

export function Step3Preferences({
  initialSubStep,
  initialPreferences,
  initialContext,
  initialDemographics,
  parseStatus = "idle",
  onRetryParse,
  onSubStepComplete,
  onComplete,
}: Step3PreferencesProps) {
  const [currentSubStep, setCurrentSubStep] = useState<Step3SubStep>(
    initialSubStep ?? "direction",
  );
  const [prefs, setPrefs] = useState<Partial<Preferences>>(
    initialPreferences ?? {},
  );
  const [ctx, setCtx] = useState<Partial<Context>>(initialContext ?? {});
  const [demo, setDemo] = useState<Partial<Demographics>>(
    initialDemographics ?? {},
  );

  const subStepIdx = SUBSTEP_ORDER.indexOf(currentSubStep);

  function goBack() {
    const prev = SUBSTEP_ORDER[subStepIdx - 1];
    if (prev) setCurrentSubStep(prev);
  }

  // BUG-018 FIX: auto-save at each substep boundary via onSubStepComplete
  function advanceWithPrefs(newPrefs: Partial<Preferences>) {
    const merged = { ...prefs, ...newPrefs };
    setPrefs(merged);
    const next = SUBSTEP_ORDER[subStepIdx + 1];
    if (next) {
      onSubStepComplete(currentSubStep, {
        preferences: merged,
        context: ctx,
        demographics: demo,
      });
      setCurrentSubStep(next);
    } else {
      onComplete({ preferences: merged, context: ctx, demographics: demo });
    }
  }

  function advanceWithCtx(newCtx: Partial<Context>) {
    const merged = { ...ctx, ...newCtx };
    setCtx(merged);
    const next = SUBSTEP_ORDER[subStepIdx + 1];
    if (next) {
      onSubStepComplete(currentSubStep, {
        preferences: prefs,
        context: merged,
        demographics: demo,
      });
      setCurrentSubStep(next);
    }
  }

  function advanceWithDemo(newDemo: Partial<Demographics>) {
    const merged = { ...demo, ...newDemo };
    setDemo(merged);
    const next = SUBSTEP_ORDER[subStepIdx + 1];
    if (next) {
      onSubStepComplete(currentSubStep, {
        preferences: prefs,
        context: ctx,
        demographics: merged,
      });
      setCurrentSubStep(next);
    } else {
      onComplete({ preferences: prefs, context: ctx, demographics: merged });
    }
  }

  return (
    <div style={pageStyle}>
      <div style={{ ...cardStyle }}>
        {/* Parse status banner — visible on all substeps */}
        {parseStatus !== "idle" && (
          <ParseBanner status={parseStatus} onRetry={onRetryParse} />
        )}

        {/* Substep content */}
        {currentSubStep === "direction" && (
          <DirectionStep
            ctx={ctx}
            onNext={advanceWithCtx}
            onBack={subStepIdx > 0 ? goBack : undefined}
          />
        )}

        {currentSubStep === "worktype" && (
          <WorkTypeStep
            prefs={prefs}
            onNext={advanceWithPrefs}
            onBack={goBack}
          />
        )}

        {currentSubStep === "comp" && (
          <CompStep prefs={prefs} onNext={advanceWithPrefs} onBack={goBack} />
        )}

        {currentSubStep === "location" && (
          <LocationStep
            prefs={prefs}
            onNext={advanceWithPrefs}
            onBack={goBack}
          />
        )}

        {currentSubStep === "quickcheck" && (
          <QuickCheckStep
            demo={demo}
            onNext={advanceWithDemo}
            onBack={goBack}
          />
        )}

        {currentSubStep === "dealbreakers" && (
          <DealBreakersStep
            prefs={prefs}
            locationTypes={prefs.locationTypes ?? ["Remote"]}
            onNext={(newPrefs) => {
              const merged = { ...prefs, ...newPrefs };
              setPrefs(merged);
              onComplete({
                preferences: merged,
                context: ctx,
                demographics: demo,
              });
            }}
            onBack={goBack}
          />
        )}
      </div>
    </div>
  );
}

        value={entry.notable}
        onChange={(v) => setField("notable", v)}
        placeholder="e.g. Published research, thesis title"
      />
      <div style={{ display: "flex", gap: 8, justifyContent: "flex-end" }}>
        <Btn variant="ghost" size="sm" onClick={onCancel}>
          Cancel
        </Btn>
        <Btn variant="primary" size="sm" onClick={handleSave}>
          Save
        </Btn>
      </div>
    </div>
  );
}

// ── Chip (de-selectable) ──────────────────────────────────────────────────────

function Chip({
  label,
  active,
  onToggle,
}: {
  label: string;
  active: boolean;
  onToggle: () => void;
}) {
  return (
    <Btn
      variant="chip"
      size="sm"
      selected={active}
      onClick={onToggle}
      aria-pressed={active}
      style={{
        borderRadius: "var(--radius-full)",
        background: active ? "var(--surface-alt)" : "transparent",
        color: active ? "var(--text)" : "var(--text-muted)",
        border: `1px solid ${active ? "var(--border)" : "transparent"}`,
        opacity: active ? 1 : 0.45,
        textDecoration: active ? "none" : "line-through",
        transition: "all 0.15s",
      }}
    >
      {label}
    </Btn>
  );
}

// ── Section wrapper ───────────────────────────────────────────────────────────

function Section({
  title,
  subtitle,
  children,
}: {
  title: string;
  subtitle?: string;
  children: React.ReactNode;
}) {
  return (
    <div
      style={{
        display: "flex",
        flexDirection: "column",
        gap: 12,
      }}
    >
      <div>
        <h3
          style={{
            fontSize: 16,
            fontWeight: 700,
            color: "var(--text)",
            margin: "0 0 2px 0",
          }}
        >
          {title}
        </h3>
        {subtitle && (
          <p style={{ fontSize: 13, color: "var(--text-muted)", margin: 0 }}>
            {subtitle}
          </p>
        )}
      </div>
      {children}
    </div>
  );
}

// ── Step4Profile ──────────────────────────────────────────────────────────────

export function Step4Profile({
  resumeStructured,
  personal: personalProp,
  education: educationProp,
  context,
  preferences,
  parseStatus,
  existingProfile,
  backgroundProfile,
  backgroundGenStatus = "idle",
  backgroundGenPhase = 0,
  backgroundGenError,
  onRetryParse,
  onComplete,
  onUploadDifferent,
}: Step4ProfileProps) {
  // ── Personal info local state + touched guard (GS-ONB-34) ────────────────
  const [personal, setPersonal] = useState<Personal>(
    personalProp ?? EMPTY_PERSONAL,
  );
  const [personalTouched, setPersonalTouched] = useState(false);

  // ── Education local state + touched guard (GS-ONB-34) ────────────────────
  const [education, setEducation] = useState<EducationEntry[]>(
    educationProp ?? [],
  );
  const [educationTouched, setEducationTouched] = useState(false);
  const [editingEduIdx, setEditingEduIdx] = useState<number | null>(null);
  const [addingEdu, setAddingEdu] = useState(false);

  // ── Profile state ─────────────────────────────────────────────────────────
  const [localProfile, setLocalProfile] = useState<Profile | null>(
    existingProfile ?? backgroundProfile ?? null,
  );
  const [genStatus, setGenStatus] = useState<Step4ProfileGenStatus>(() => {
    if (existingProfile || backgroundProfile) return "done";
    return backgroundGenStatus;
  });
  const [genPhase, setGenPhase] = useState<number>(backgroundGenPhase);
  const [genError, setGenError] = useState<string | null>(
    backgroundGenError ?? null,
  );

  // ── Profile display state (editable fields) ───────────────────────────────
  const [discipline, setDiscipline] = useState(localProfile?.discipline ?? "");
  const [seniority, setSeniority] = useState(localProfile?.seniority ?? "");
  const [targetDirection, setTargetDirection] = useState(
    localProfile?.targetDirection ?? "",
  );
  const [activeSkills, setActiveSkills] = useState<Set<string>>(
    new Set(localProfile?.hardSkills ?? []),
  );
  const [activeDomains, setActiveDomains] = useState<Set<string>>(
    new Set(localProfile?.domains ?? []),
  );

  // ── Validation errors ─────────────────────────────────────────────────────
  const [personalErrors, setPersonalErrors] = useState<Record<string, string>>(
    {},
  );
  const [portfolioError, setPortfolioError] = useState<string | null>(null);

  // ── AbortController for generation ───────────────────────────────────────
  const abortRef = useRef<AbortController | null>(null);
  const phaseTimerRef = useRef<ReturnType<typeof setInterval> | null>(null);

  // ── GS-ONB-34: Sync personal from props when late parse arrives ───────────
  useEffect(() => {
    if (!personalTouched && personalProp?.name) {
      setPersonal(personalProp);
    }
  }, [personalProp]); // eslint-disable-line react-hooks/exhaustive-deps

  // ── GS-ONB-34: Sync education from props when late parse arrives ──────────
  useEffect(() => {
    if (!educationTouched && educationProp?.length) {
      setEducation(educationProp);
    }
  }, [educationProp]); // eslint-disable-line react-hooks/exhaustive-deps

  // ── GS-ONB-33: React to backgroundGenStatus prop changes ─────────────────
  useEffect(() => {
    if (backgroundGenStatus === "done" && backgroundProfile && !localProfile) {
      setLocalProfile(backgroundProfile);
      setGenStatus("done");
      applyProfile(backgroundProfile);
    } else if (backgroundGenStatus === "generating") {
      if (genStatus !== "done") {
        setGenStatus("generating");
        setGenPhase(backgroundGenPhase);
      }
    } else if (backgroundGenStatus === "error" && genStatus !== "done") {
      setGenStatus("error");
      setGenError(backgroundGenError ?? "Profile generation failed.");
    }
  }, [
    backgroundGenStatus,
    backgroundProfile,
    backgroundGenPhase,
    backgroundGenError,
  ]); // eslint-disable-line react-hooks/exhaustive-deps

  // ── Auto-trigger generation if background gen didn't run ─────────────────
  useEffect(() => {
    if (
      genStatus === "idle" &&
      !localProfile &&
      parseStatus === "done" &&
      resumeStructured
    ) {
      triggerGeneration();
    }
  }, []); // eslint-disable-line react-hooks/exhaustive-deps

  // ── Cleanup on unmount ────────────────────────────────────────────────────
  useEffect(() => {
    return () => {
      abortRef.current?.abort();
      if (phaseTimerRef.current) clearInterval(phaseTimerRef.current);
    };
  }, []);

  function applyProfile(p: Profile) {
    setDiscipline(p.discipline ?? "");
    setSeniority(p.seniority ?? "");
    setTargetDirection(p.targetDirection ?? "");
    setActiveSkills(new Set(p.hardSkills ?? []));
    setActiveDomains(new Set(p.domains ?? []));
  }

  function startPhaseTicker() {
    let phase = 0;
    setGenPhase(0);
    if (phaseTimerRef.current) clearInterval(phaseTimerRef.current);
    phaseTimerRef.current = setInterval(() => {
      phase = Math.min(phase + 1, PROFILE_PHASES.length - 1);
      setGenPhase(phase);
    }, 5000);
  }

  function stopPhaseTicker() {
    if (phaseTimerRef.current) {
      clearInterval(phaseTimerRef.current);
      phaseTimerRef.current = null;
    }
  }

  const triggerGeneration = useCallback(async () => {
    if (genStatus === "generating") return;

    // Cancel any in-flight generation
    abortRef.current?.abort();
    abortRef.current = new AbortController();

    setGenStatus("generating");
    setGenError(null);
    startPhaseTicker();

    const profileInput = JSON.stringify({
      resume: resumeStructured,
      personal: personalProp,
      context: {
        empStatus: context.empStatus ?? "employed",
        careerDirection:
          context.careerDirection ?? context.careerDirectionText ?? "",
        directionPresets: context.directionPresets ?? [],
        whyLeft: context.whyLeft ?? "",
        latestRoleText: context.latestRoleText ?? "",
      },
      preferences: {
        locationTypes: preferences.locationTypes ?? ["Remote"],
        employmentTypes: preferences.employmentTypes ?? ["Full-time"],
        compStructure: preferences.compStructure ?? "Open to discussion",
        salaryFloor: preferences.salaryFloor ?? "",
        hourlyFloor: preferences.hourlyFloor ?? "",
        startDate: preferences.startDate ?? "Immediately",
        dealBreakers: preferences.dealBreakers ?? [],
      },
    });

    try {
      const raw = await callClaude(
        "PROFILE",
        profileInput,
        4096,
        abortRef.current.signal,
      );
      stopPhaseTicker();

      const parsed = JSON.parse(cleanJson(raw)) as Profile;
      setLocalProfile(parsed);
      applyProfile(parsed);
      setGenStatus("done");
    } catch (err) {
      stopPhaseTicker();
      if (err instanceof Error && err.message.includes("aborted")) return;
      setGenStatus("error");
      setGenError(
        safeErrorMessage(err, "Profile generation failed. Please try again."),
      );
    }
  }, [resumeStructured, personalProp, context, preferences]); // eslint-disable-line react-hooks/exhaustive-deps

  // ── Personal field updater ────────────────────────────────────────────────
  function updatePersonal(field: keyof Personal, value: string) {
    setPersonalTouched(true);
    setPersonal((prev) => ({ ...prev, [field]: value }));
    setPersonalErrors((prev) => ({ ...prev, [field]: "" }));
    if (field === "portfolio") setPortfolioError(null);
  }

  // ── Confirm + validate ────────────────────────────────────────────────────
  function handleConfirm() {
    if (!localProfile) return;

    const errs: Record<string, string> = {};
    if (!personal.name.trim()) errs.name = "Name is required";
    if (!personal.email.trim()) errs.email = "Email is required";
    if (!personal.phone.trim()) errs.phone = "Phone is required";
    if (!personal.location.trim()) errs.location = "Location is required";

    const portErr = validatePortfolio(personal.portfolio ?? "");
    if (portErr) {
      setPortfolioError(portErr);
      return;
    }

    if (Object.keys(errs).length) {
      setPersonalErrors(errs);
      return;
    }

    // Build final profile with user overrides
    const finalProfile: Profile = {
      ...localProfile,
      discipline,
      seniority,
      targetDirection,
      hardSkills: (localProfile.hardSkills ?? []).filter((s) =>
        activeSkills.has(s),
      ),
      domains: (localProfile.domains ?? []).filter((d) => activeDomains.has(d)),
    };

    // GS-ONB-24: context defaults (no empStatus etc on confirm screen)
    const finalContext: Partial<Context> = {
      ...context,
      empStatus: context.empStatus ?? "employed",
      notOnResume: context.notOnResume ?? "",
      titleNotCapture: context.titleNotCapture ?? "",
    };

    onComplete({
      profile: finalProfile,
      personal,
      education,
      context: finalContext,
    });
  }

  // ── Shared page layout ────────────────────────────────────────────────────
  const outerStyle: React.CSSProperties = {
    display: "flex",
    flexDirection: "column",
    alignItems: "center",
    padding: "32px 24px 80px",
    background: "var(--bg)",
    minHeight: "100%",
  };
  const contentStyle: React.CSSProperties = {
    display: "flex",
    flexDirection: "column",
    width: "100%",
    maxWidth: 600,
    gap: 32,
  };

  // ── GS-ONB-36: Waiting for parse ──────────────────────────────────────────
  if (parseStatus === "parsing") {
    return (
      <div style={outerStyle}>
        <div
          style={{
            ...contentStyle,
            alignItems: "center",
            gap: 20,
            paddingTop: 60,
          }}
        >
          <Spin size="lg" />
          <div style={{ textAlign: "center" }}>
            <h2
              style={{
                fontSize: 20,
                fontWeight: 700,
                color: "var(--text)",
                margin: "0 0 8px 0",
              }}
            >
              Waiting for resume parsing to complete…
            </h2>
            <p style={{ fontSize: 14, color: "var(--text-muted)", margin: 0 }}>
              Profile generation will start as soon as parsing completes.
            </p>
            <div
              style={{
                marginTop: 16,
                display: "flex",
                flexDirection: "column",
                gap: 4,
              }}
            >
              {[
                "Reading document…",
                "Extracting structure…",
                "Building profile…",
              ].map((p) => (
                <p
                  key={p}
                  style={{
                    fontSize: 13,
                    color: "var(--text-muted)",
                    margin: 0,
                  }}
                >
                  {p}
                </p>
              ))}
            </div>
          </div>
        </div>
      </div>
    );
  }

  // ── GS-ONB-37: Parse error recovery ──────────────────────────────────────
  if (parseStatus === "error") {
    return (
      <div style={outerStyle}>
        <div
          style={{
            ...contentStyle,
            alignItems: "center",
            gap: 20,
            paddingTop: 60,
          }}
        >
          <div
            style={{
              background: "var(--error-light)",
              border: "1px solid var(--error-border)",
              borderRadius: "var(--radius-lg)",
              padding: "24px",
              textAlign: "center",
              width: "100%",
            }}
          >
            <p
              style={{
                fontSize: 16,
                fontWeight: 600,
                color: "var(--error)",
                margin: "0 0 8px 0",
              }}
            >
              Resume parsing failed.
            </p>
            <p
              style={{
                fontSize: 14,
                color: "var(--text-muted)",
                margin: "0 0 16px 0",
              }}
            >
              We couldn&apos;t extract your resume data. Try again or upload a
              different file.
            </p>
            <div style={{ display: "flex", gap: 10, justifyContent: "center" }}>
              {onRetryParse && (
                <Btn
                  variant="primary"
                  onClick={onRetryParse}
                  aria-label="Retry resume parsing"
                >
                  Retry
                </Btn>
              )}
              {onUploadDifferent && (
                <Btn
                  variant="ghost"
                  onClick={onUploadDifferent}
                  aria-label="Upload a different resume"
                >
                  Upload different
                </Btn>
              )}
            </div>
          </div>
        </div>
      </div>
    );
  }

  // ── Profile generating ────────────────────────────────────────────────────
  if (genStatus === "generating") {
    return (
      <div style={outerStyle}>
        <div
          style={{
            ...contentStyle,
            alignItems: "center",
            gap: 24,
            paddingTop: 60,
          }}
        >
          <Spin size="lg" />
          <div style={{ textAlign: "center" }}>
            <h2
              style={{
                fontSize: 20,
                fontWeight: 700,
                color: "var(--text)",
                margin: "0 0 16px 0",
              }}
            >
              Building your profile…
            </h2>
            <div style={{ display: "flex", flexDirection: "column", gap: 6 }}>
              {PROFILE_PHASES.map((phase, i) => (
                <p
                  key={phase}
                  style={{
                    fontSize: 14,
                    color:
                      i === genPhase
                        ? "var(--text)"
                        : i < genPhase
                          ? "var(--success)"
                          : "var(--border)",
                    fontWeight: i === genPhase ? 600 : 400,
                    margin: 0,
                    transition: "color 0.3s",
                  }}
                >
                  {i < genPhase ? "✓ " : ""}
                  {phase}
                </p>
              ))}
            </div>
          </div>
        </div>
      </div>
    );
  }

  // ── GS-ONB-38: Generation error ───────────────────────────────────────────
  if (genStatus === "error") {
    return (
      <div style={outerStyle}>
        <div
          style={{
            ...contentStyle,
            alignItems: "center",
            gap: 20,
            paddingTop: 60,
          }}
        >
          <div
            style={{
              background: "var(--error-light)",
              border: "1px solid var(--error-border)",
              borderRadius: "var(--radius-lg)",
              padding: "24px",
              textAlign: "center",
              width: "100%",
            }}
          >
            <p
              style={{
                fontSize: 15,
                color: "var(--error)",
                margin: "0 0 8px 0",
                fontWeight: 600,
              }}
            >
              Profile generation failed.
            </p>
            {genError && (
              <p
                style={{
                  fontSize: 13,
                  color: "var(--text-muted)",
                  margin: "0 0 16px 0",
                }}
              >
                {genError}
              </p>
            )}
            <Btn
              variant="primary"
              onClick={triggerGeneration}
              aria-label="Retry profile generation"
            >
              Retry generation
            </Btn>
          </div>
        </div>
      </div>
    );
  }

  // ── genStatus === 'idle' and no profile yet ───────────────────────────────
  if (genStatus === "idle" && !localProfile) {
    return (
      <div style={outerStyle}>
        <div style={{ ...contentStyle, alignItems: "center", paddingTop: 60 }}>
          <Spin size="lg" />
        </div>
      </div>
    );
  }

  // ── Full review form ──────────────────────────────────────────────────────

  const profile = localProfile!;

  const friendlyName = personal.name ? `, ${personal.name.split(" ")[0]}` : "";

  return (
    <div style={outerStyle}>
      <div style={contentStyle}>
        {/* Friendly intro */}
        <div>
          <h2
            style={{
              fontSize: 22,
              fontWeight: 700,
              color: "var(--text)",
              margin: "0 0 6px 0",
            }}
          >
            Confirm your info
          </h2>
          <p style={{ fontSize: 14, color: "var(--text-muted)", margin: 0 }}>
            Thanks{friendlyName}. Fix anything that&apos;s off — this powers
            everything we build for you.
          </p>
        </div>

        {/* ── Section: Personal ─────────────────────────────────────────── */}
        <Section
          title="Personal"
          subtitle="We extracted this from your resume. Edit anything that looks off."
        >
          <Inp
            label="Name"
            value={personal.name}
            onChange={(v) => updatePersonal("name", v)}
            required
            error={personalErrors.name}
          />
          <Inp
            label="Email"
            value={personal.email}
            onChange={(v) => updatePersonal("email", v)}
            type="email"
            required
            error={personalErrors.email}
          />
          <Inp
            label="Phone"
            value={personal.phone}
            onChange={(v) => updatePersonal("phone", v)}
            required
            error={personalErrors.phone}
          />
          <Inp
            label="Location"
            value={personal.location}
            onChange={(v) => updatePersonal("location", v)}
            required
            error={personalErrors.location}
          />
          <Inp
            label="LinkedIn URL"
            value={personal.linkedin}
            onChange={(v) => updatePersonal("linkedin", v)}
            placeholder="https://linkedin.com/in/yourname"
          />
          <div>
            <Inp
              label="Portfolio URL (optional)"
              value={personal.portfolio ?? ""}
              onChange={(v) => {
                updatePersonal("portfolio", v);
                setPortfolioError(null);
              }}
              placeholder="https://yoursite.com"
              error={portfolioError ?? undefined}
            />
          </div>
        </Section>

        {/* ── Section: Education ────────────────────────────────────────── */}
        <Section title="Education">
          {education.length === 0 && !addingEdu && (
            <p style={{ fontSize: 14, color: "var(--text-muted)" }}>
              No education found on your resume. Want to add one?
            </p>
          )}

          {education.map((edu, idx) =>
            editingEduIdx === idx ? (
              <EduForm
                key={idx}
                initial={edu}
                onSave={(updated) => {
                  setEducationTouched(true);
                  setEducation((prev) =>
                    prev.map((e, i) => (i === idx ? updated : e)),
                  );
                  setEditingEduIdx(null);
                }}
                onCancel={() => setEditingEduIdx(null)}
              />
            ) : (
              <EduCard
                key={idx}
                education={edu}
                onEdit={() => setEditingEduIdx(idx)}
                onRemove={() => {
                  setEducationTouched(true);
                  setEducation((prev) => prev.filter((_, i) => i !== idx));
                }}
              />
            ),
          )}

          {addingEdu && (
            <EduForm
              onSave={(entry) => {
                setEducationTouched(true);
                setEducation((prev) => [...prev, entry]);
                setAddingEdu(false);
              }}
              onCancel={() => setAddingEdu(false)}
            />
          )}

          {!addingEdu && editingEduIdx === null && (
            <Btn
              variant="ghost"
              size="sm"
              onClick={() => setAddingEdu(true)}
              style={{ alignSelf: "flex-start" }}
              aria-label="Add education entry"
            >
              + Add education
            </Btn>
          )}
        </Section>

        {/* ── Section: Who you are (AI Profile) ────────────────────────── */}
        <Section title="Who you are" subtitle="Your professional identity.">
          {/* GS-ONB-38: show error box in profile section if gen failed after partial use */}

          {/* GS-ONB-19: Discipline MUST be free-text Inp, NOT Sel */}
          <Inp
            label="Discipline"
            value={discipline}
            onChange={setDiscipline}
            placeholder="e.g. Product Manager"
          />
          <Sel
            label="Seniority"
            value={seniority}
            onChange={setSeniority}
            options={SENIORITY_OPTIONS.map((s) => ({ value: s, label: s }))}
            placeholder="Select seniority"
          />
        </Section>

        {/* ── Section: Skills ───────────────────────────────────────────── */}
        {(profile.hardSkills?.length ?? 0) > 0 && (
          <Section
            title="Your skills"
            subtitle="We'll use these to match you with jobs."
          >
            <div style={{ display: "flex", flexWrap: "wrap", gap: 6 }}>
              {profile.hardSkills?.map((skill) => (
                <Chip
                  key={skill}
                  label={skill}
                  active={activeSkills.has(skill)}
                  onToggle={() =>
                    setActiveSkills((prev) => {
                      const next = new Set(prev);
                      if (next.has(skill)) next.delete(skill);
                      else next.add(skill);
                      return next;
                    })
                  }
                />
              ))}
            </div>
            <p style={{ fontSize: 12, color: "var(--text-muted)", margin: 0 }}>
              Extracted {activeSkills.size} skills
              {profile.differentiators
                ? ` and ${profile.differentiators.length} differentiators`
                : ""}{" "}
              from your resume. You&apos;ll curate these after the market scan.
            </p>
          </Section>
        )}

        {/* ── Section: Domains ──────────────────────────────────────────── */}
        {(profile.domains?.length ?? 0) > 0 && (
          <Section title="Domains">
            <div style={{ display: "flex", flexWrap: "wrap", gap: 6 }}>
              {profile.domains?.map((d) => (
                <Chip
                  key={d}
                  label={d}
                  active={activeDomains.has(d)}
                  onToggle={() =>
                    setActiveDomains((prev) => {
                      const next = new Set(prev);
                      if (next.has(d)) next.delete(d);
                      else next.add(d);
                      return next;
                    })
                  }
                />
              ))}
            </div>
          </Section>
        )}

        {/* ── Section: What makes you stand out ────────────────────────── */}
        {(profile.differentiators?.length ?? 0) > 0 && (
          <Section
            title="What makes you stand out"
            subtitle="Your competitive edge."
          >
            <ul
              style={{
                margin: 0,
                padding: "0 0 0 18px",
                display: "flex",
                flexDirection: "column",
                gap: 6,
              }}
            >
              {profile.differentiators?.map((d, i) => (
                <li key={i} style={{ fontSize: 14, color: "var(--text)" }}>
                  {d}
                </li>
              ))}
            </ul>
          </Section>
        )}

        {/* ── Section: Key achievements ─────────────────────────────────── */}
        {(profile.achievements?.length ?? 0) > 0 && (
          <Section
            title="Key achievements"
            subtitle="The wins we'll highlight in your resumes."
          >
            <ul
              style={{
                margin: 0,
                padding: "0 0 0 18px",
                display: "flex",
                flexDirection: "column",
                gap: 6,
              }}
            >
              {profile.achievements?.map((a, i) => (
                <li key={i} style={{ fontSize: 14, color: "var(--text)" }}>
                  {a}
                </li>
              ))}
            </ul>
          </Section>
        )}

        {/* ── Section: Where you're headed ──────────────────────────────── */}
        <Section
          title="Where you're headed"
          subtitle="This guides which jobs we target."
        >
          <Inp
            label="Target direction"
            value={targetDirection}
            onChange={setTargetDirection}
            placeholder="e.g. Senior PM at a growth-stage startup"
          />
        </Section>

  // ── Routing ───────────────────────────────────────────────────────────────
  const [innerStep, setInnerStep] = useState<InnerStep>(1);

  const [parseStatus, setParseStatus] = useState<Step4ParseStatus>("idle");
  const [resumeStructured, setResumeStructured] = useState<ResumeStructured>(
    EMPTY_RESUME_STRUCTURED,
  );
  const parseAbortRef = useRef<AbortController | null>(null);

  // ── Background profile generation ─────────────────────────────────────────
  const [bgGenStatus, setBgGenStatus] = useState<Step4ProfileGenStatus>("idle");
  const [bgGenPhase, setBgGenPhase] = useState(0);
  const [bgGenError, setBgGenError] = useState<string | null>(null);
  const [bgProfile, setBgProfile] = useState<Profile | null>(null);
  const profileGenAbortRef = useRef<AbortController | null>(null);
  const phaseTickerRef = useRef<ReturnType<typeof setInterval> | null>(null);

  // ── Preferences substep progress ─────────────────────────────────────────
  // Tracks which dot is "active" in OnboardingProgress
  const [prefSubStep, setPrefSubStep] = useState(0);

  // ── Background profile generation ────────────────────────────────────────

  async function triggerProfileGen(parsedResume: ResumeStructured) {
    // Cancel any in-flight gen
    profileGenAbortRef.current?.abort();
    const controller = new AbortController();
    profileGenAbortRef.current = controller;

    // Clear phase ticker
    if (phaseTickerRef.current) {
      clearInterval(phaseTickerRef.current);
      phaseTickerRef.current = null;
    }

    setBgGenStatus("generating");
    setBgGenPhase(0);
    setBgProfile(null);
    setBgGenError(null);

    // Phase ticker — advances every 5s through 4 phases
    let phase = 0;
    phaseTickerRef.current = setInterval(() => {
      phase = Math.min(phase + 1, 3);
      setBgGenPhase(phase);
    }, 5000);

    // Use sessionRef.current for preferences/context — by the time gen starts,
    // the user may have already completed step 2 (full data available).
    const profileInput = JSON.stringify({
      resume: parsedResume,
      personal: parsedResume.personal ?? EMPTY_PERSONAL,
      context: sessionRef.current.context ?? {},
      preferences: sessionRef.current.preferences ?? {},
    });

    try {
      const raw = await callClaude(
        "PROFILE",
        profileInput,
        4096,
        controller.signal,
      );
      if (controller.signal.aborted) return;

      const cleaned = cleanJson(raw);
      const profile = JSON.parse(cleaned) as Profile;

      if (phaseTickerRef.current) {
        clearInterval(phaseTickerRef.current);
        phaseTickerRef.current = null;
      }

      setBgProfile(profile);
      setBgGenStatus("done");

      // Persist to session (non-blocking — don't await, not critical path)

      sessionRef.current = newSession;
      setSession(newSession);
      saveSession(newSession).catch(() => {});
    } catch (err) {

        phaseTickerRef.current = null;

    }
  }

    // Cancel any in-flight parse
    parseAbortRef.current?.abort();
    const controller = new AbortController();
    parseAbortRef.current = controller;

    setParseStatus("parsing");

    try {
      const raw = await callClaude("PARSE", resumeRaw, 4096, controller.signal);
      if (controller.signal.aborted) return;

      const cleaned = cleanJson(raw);
      const parsed = JSON.parse(cleaned) as ResumeStructured;

      setResumeStructured(parsed);
      setParseStatus("done");

      // Merge parsed data into session
      const updated: SessionData = {
        ...sessionRef.current,
        resumeStructured: parsed,
        // Carry personal + education extracted from parse (only if not already user-set)
        personal: sessionRef.current.personal ?? parsed.personal,
        education: sessionRef.current.education?.length
          ? sessionRef.current.education
          : (parsed.education ?? []),
      };
      sessionRef.current = updated;
      setSession(updated);
      await saveSession(updated);

      // Immediately kick off profile gen (runs in background while user is on step 2 or 3)
      triggerProfileGen(parsed);
    } catch (err) {
      if (controller.signal.aborted) return;
      setParseStatus("error");
      // Error is surfaced via ParseBanner — no need to store separately
      void safeErrorMessage(err);
    }
  }

  // ── Mount: load session and route to correct step ─────────────────────────

  useEffect(() => {
    loadSession().then((s) => {
      const loaded = s ?? {};
      sessionRef.current = loaded;
      setSession(loaded);

      const step = loaded.currentStep ?? 1;

      if (step >= 3) {
        // User is at step 3 (profile) — already has parsed resume
        setInnerStep(3);
        if (loaded.resumeStructured) {
          setResumeStructured(loaded.resumeStructured);
        }
        // If bg gen was already persisted (previous session), treat as done
        // Step4Profile will receive existingProfile and skip re-gen
        setParseStatus("done");

        // User is at step 2 (preferences)
        setInnerStep(2);

          // Parse already done in a previous session
          setResumeStructured(loaded.resumeStructured);
          setParseStatus("done");
          // Re-trigger profile gen if it wasn't completed yet
          if (!loaded.profile) {
            triggerProfileGen(loaded.resumeStructured);
          }
        } else if (loaded.resumeRaw) {
          // Parse didn't complete — re-trigger it
          triggerParse(loaded.resumeRaw);
        }
      } else {
        // Default to step 1
        setInnerStep(1);
      }

      setSessionLoaded(true);
    });

    return () => {
      // Cleanup all in-flight async work on unmount
      parseAbortRef.current?.abort();
      profileGenAbortRef.current?.abort();
      if (phaseTickerRef.current) clearInterval(phaseTickerRef.current);
    };
  }, []); // eslint-disable-line react-hooks/exhaustive-deps

// src/lib/upload.ts — Client-side resume file extraction helpers
// BUG-020 FIX: DOCX extraction uses JSZip for proper deflate decompression.

export type ValidateResult = { valid: true } | { valid: false; error: string };

// ── Constants (from VALIDATION_RULES.md) ─────────────────────────────────────

const MAX_FILE_SIZE = 10 * 1024 * 1024; // 10 MB
const MAX_PDF_PAGES = 50;
const MAX_EXTRACTED_TEXT = 500_000; // chars
const PARSE_TIMEOUT_MS = 15_000;

const ALLOWED_EXTENSIONS = new Set([".pdf", ".docx", ".txt", ".md"]);

function getExtension(filename: string): string {
  const lastDot = filename.lastIndexOf(".");
  if (lastDot === -1) return "";
  return filename.slice(lastDot).toLowerCase();
}

// ── validateFile (synchronous — used by page.tsx and components) ──────────────

export function validateFile(file: File): ValidateResult {
  if (file.size === 0) {
    return { valid: false, error: "File is empty." };
  }
  const sizeMb = (file.size / (1024 * 1024)).toFixed(1);
  if (file.size > MAX_FILE_SIZE) {
    return {
      valid: false,
      error: `File too large (${sizeMb}MB). Maximum is 10MB.`,
    };
  }
  const ext = getExtension(file.name);
  if (!ALLOWED_EXTENSIONS.has(ext)) {
    return {
      valid: false,
      error: "Unsupported file type. Accepted: PDF, DOCX, TXT, MD.",
    };
  }
  return { valid: true };
}

// ── validateFileDeep (async — adds magic byte check) ─────────────────────────

export async function validateFileDeep(file: File): Promise<ValidateResult> {
  const basic = validateFile(file);
  if (!basic.valid) return basic;

  const ext = getExtension(file.name);

  // Read first 8 bytes for magic byte header check
  const slice = file.slice(0, 8);
  const buffer = await slice.arrayBuffer();
  const bytes = new Uint8Array(buffer);

  if (ext === ".pdf") {
    // %PDF = 0x25 0x50 0x44 0x46
    if (
      bytes[0] !== 0x25 ||
      bytes[1] !== 0x50 ||
      bytes[2] !== 0x44 ||
      bytes[3] !== 0x46
    ) {
      return { valid: false, error: "File does not appear to be a valid PDF." };
    }
  } else if (ext === ".docx") {
    // PK = 0x50 0x4B 0x03 0x04 (ZIP local file header)
    if (
      bytes[0] !== 0x50 ||
      bytes[1] !== 0x4b ||
      bytes[2] !== 0x03 ||
      bytes[3] !== 0x04
    ) {
      return {
        valid: false,
        error: "File does not appear to be a valid DOCX.",
      };
    }
  }

  return { valid: true };
}

// ── extractText (deep-validate + parse, with timeout) ────────────────────────

export async function extractText(file: File): Promise<string> {
  const check = await validateFileDeep(file);
  if (!check.valid) throw new Error(check.error);

  const ext = getExtension(file.name);

  let extractPromise: Promise<string>;
  if (ext === ".pdf") {
    extractPromise = extractPdfText(file);
  } else if (ext === ".docx") {
    extractPromise = extractDocxText(file);
  } else {
    extractPromise = extractPlainText(file);
  }

  const timeoutPromise = new Promise<never>((_, reject) =>
    setTimeout(
      () => reject(new Error("File parsing timed out (15s).")),
      PARSE_TIMEOUT_MS,
    ),
  );

  const text = await Promise.race([extractPromise, timeoutPromise]);

  if (!text || text.trim().length === 0) {
    throw new Error("No text could be extracted from the file.");
  }

  return text.length > MAX_EXTRACTED_TEXT
    ? text.slice(0, MAX_EXTRACTED_TEXT)
    : text;
}

// ── PDF extraction via pdfjs-dist ─────────────────────────────────────────────

async function extractPdfText(file: File): Promise<string> {
  // Dynamic import to keep pdfjs out of SSR / initial bundle
  const pdfjsLib = await import("pdfjs-dist");

  // Configure worker using CDN; safe to set multiple times (idempotent)
  if (typeof window !== "undefined") {
    const ver =
      (pdfjsLib as unknown as { version?: string }).version ?? "5.6.205";
    pdfjsLib.GlobalWorkerOptions.workerSrc = `https://cdn.jsdelivr.net/npm/pdfjs-dist@${ver}/build/pdf.worker.min.mjs`;
  }

  const buffer = await file.arrayBuffer();
  const pdf = await pdfjsLib.getDocument({ data: buffer }).promise;

  const pageCount = Math.min(pdf.numPages, MAX_PDF_PAGES);
  const pages: string[] = [];

  for (let i = 1; i <= pageCount; i++) {
    const page = await pdf.getPage(i);
    const content = await page.getTextContent();
    const pageText = (content.items as Array<{ str?: string }>)
      .filter((item) => typeof item.str === "string" && item.str.trim() !== "")
      .map((item) => item.str as string)
      .join(" ");
    if (pageText.trim()) pages.push(pageText);
  }

  return pages.join("\n\n");
}

// ── DOCX extraction via JSZip (BUG-020 fix: proper ZIP/deflate decompression) ─

async function extractDocxText(file: File): Promise<string> {
  // Dynamic import: JSZip handles ZIP/deflate natively in browser & Node
  const JSZipModule = await import("jszip");
  const JSZip =
    (JSZipModule as unknown as { default?: typeof JSZipModule }).default ??
    JSZipModule;

  const buffer = await file.arrayBuffer();
  // JSZip.loadAsync handles deflate decompression correctly (fixes BUG-020)
  const zip = await (
    JSZip as unknown as {
      loadAsync: (
        b: ArrayBuffer,
      ) => Promise<{
        file: (
          name: string,
        ) => { async: (type: string) => Promise<string> } | null;
      }>;
    }
  ).loadAsync(buffer);

  const docFile = zip.file("word/document.xml");
  if (!docFile) {
    throw new Error("Invalid DOCX: word/document.xml not found.");
  }

  const xml = await docFile.async("string");

  // Extract readable text from WordprocessingML XML
  const text = xml
    .replace(/<w:p[ >\/]/g, "\n") // paragraph → newline
    .replace(/<w:br[^>]*\/>/g, "\n") // line break → newline
    .replace(/<w:tab\s*\/>/g, "\t") // tab character
    .replace(/<[^>]+>/g, "") // strip remaining XML tags
    .replace(/&amp;/g, "&")
    .replace(/&lt;/g, "<")
    .replace(/&gt;/g, ">")
    .replace(/&quot;/g, '"')
    .replace(/&#39;|&apos;/g, "'")
    .replace(/\n{3,}/g, "\n\n")
    .trim();

  return text;
}

// ── Plain text extraction ─────────────────────────────────────────────────────

function extractPlainText(file: File): Promise<string> {
  return new Promise<string>((resolve, reject) => {
    const reader = new FileReader();
    reader.onload = (e) => {
      const result = e.target?.result;
      resolve(typeof result === "string" ? result : "");
    };
    reader.onerror = () => reject(new Error("Failed to read file."));
    reader.readAsText(file, "UTF-8");
  });
}
